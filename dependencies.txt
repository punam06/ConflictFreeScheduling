# C++ Conflict-Free Scheduling System Dependencies

# Required Build Tools:
# - g++ >= 9.0 (or clang++ >= 10.0)
# - cmake >= 3.16
# - make
# - sqlite3 development libraries
# - pkg-config

# Optional Development Tools:
# - clang-format (for code formatting)
# - cppcheck (for static analysis) 
# - valgrind (for memory leak detection)
# - gdb (for debugging)
# - doxygen (for documentation generation)

# Optional Visualization:
# - gnuplot (for performance graphs)

# Database Options:
# - sqlite3 (required - lightweight embedded database)
# - postgresql-dev (optional - for production PostgreSQL support)
# - mysql-dev (optional - for MySQL support)

# Installation commands for different systems:

# Ubuntu/Debian:
# sudo apt-get update
# sudo apt-get install build-essential cmake clang-format cppcheck valgrind gdb doxygen gnuplot
# sudo apt-get install libsqlite3-dev pkg-config libpq-dev

# macOS (with Homebrew):
# brew install cmake clang-format cppcheck valgrind doxygen gnuplot
# brew install sqlite3 pkg-config postgresql

# Fedora/RHEL:
# sudo dnf install gcc-c++ cmake clang-tools-extra cppcheck valgrind gdb doxygen gnuplot
# sudo dnf install sqlite-devel pkgconfig postgresql-devel

# Arch Linux:
# sudo pacman -S base-devel cmake clang cppcheck valgrind gdb doxygen gnuplot
# sudo pacman -S sqlite pkgconf postgresql-libs
